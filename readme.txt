python manage.py runserver - команда для запуска сервера

python manage.py makemigrations — создание миграций 
python manage.py migrate — применить созданные миграции
python manage.py showmigrations — показать все изменения вносимые в базу данных

python manage.py createsuperuser - создать админа
python manage.py changepassword <имя пользователя> - сбросить пароль админа
